[docs]    def is_involution(self):
        """Determine if the permutation is an involution, i.e., is equal to it's own inverse."""
        for idx, val in enumerate(self):
            if idx != self[val]:
                return False
        return True



[docs]    def is_increasing(self):
        """Determine if the permutation is increasing.

        Examples:
            >>> p = Permutation.random(10)
            >>> (p * p.inverse()).is_increasing()
            True

        """
        return all(idx == val for idx, val in enumerate(self))



[docs]    def is_decreasing(self):
        """Determine if the permutation is increasing.

        Examples:
            >>> p = Permutation(range(10,0,-1))
            >>> p.is_decreasing()
            True

        """
        return all(idx == val for idx, val in enumerate(self[::-1]))



[docs]    def is_identity(self):
